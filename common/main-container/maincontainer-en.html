Main Container

Main Container with "article"

Main Container with "section"


Code


	<h2>Main Container</h2>
    <main property="mainContentPage">

        <!-- heading level 1 -->
    
        <!-- content page -->
    
        <!-- Page details section [Must be configured as a <section> element in this situation] -->
    
    </main>

    <h2>Main Container with "article"</h2>
    <main property="mainContentPage">
        <article>
        <!-- heading level 1 -->
    
        <!-- content page -->
    
        <!-- Page details section [Must be configured as a <footer> element in this situation] -->
    
        </article>
    </main>

    <h2>Main Container with "section"</h2>
    <main property="mainContentPage">
        <section>
        <!-- heading level 1 -->
    
        <!-- content page -->
    
        <!-- Page details section [Must be configured as a <footer> element in this situation] -->
    
        </section>
    </main>
